\begin{enumerate}
	\item Placer les points $B$ et $D$ tels que $\vec{AB}\begin{pmatrix}1 \\ 2\end{pmatrix}$ et $\vec{CD}\begin{pmatrix}2 \\ -3\end{pmatrix}$.
	\item Construire :

	      \begin{itemize}
		      \item Un vecteur $\vec{AB'}$, tel que $∥\vec{AB}∥ = ∥\vec{AB'}∥$ et la droite $(AB)$ est perpendiculaire à $(AB')$.
		      \item Un vecteur $\vec{CD'}$, tel que $∥\vec{CD}∥ = ∥\vec{CD'}∥$ et la droite $(CD)$ est perpendiculaire à $(CD')$.
	      \end{itemize}
	\item Quels sont alors les coordonnées de $\vec{AB'}$ et $\vec{CD'}$ ? $\vec{AB'}\begin{pmatrix}\correctionOr{\color{red}-2}{\phantom{333}} \\ \correctionOr{\color{red}1}{\phantom{333}}\end{pmatrix}$ $\vec{CD'}\begin{pmatrix}\correctionOr{\color{red}-3}{\phantom{333}} \\ \correctionOr{\color{red}-2}{\phantom{333}}\end{pmatrix}$

	      Que remarque-t'on par rapport aux coordonnées de $\vec{AB}$ et $\vec{CD}$ ?

	      \correction{Les coordonnées horizontales et verticales sont échangées, et l'une d'entre elles est l'opposé.}
\end{enumerate}
