\begin{tcolorbox}
	On rappelle les identités remarquables :
	\begin{center}
		\begin{enumerate}
			\item[\circled{1}] $(a + b)² = a² + 2ab + b²$
			\item[\circled{2}] $(a - b)² = a² - 2ab + b²$
			\item[\circled{3}] $(a + b)(a - b) = a² - b²$
		\end{enumerate}
	\end{center}
\end{tcolorbox}

\begin{exercice}
	Écrire les expressions suivantes sous forme factorisée. Si une identité remarquable est utilisée, indiquer laquelle. \medskip
	\begin{enumerate}
		\setlength{\itemsep}{1em}
		\item $3x² + 2x = \correctionOr{x}{\myDots} × \correctionOr{(3x + 2)}{\myDots}$
		\item $x² + 2x + 1 = \correctionOr{(x + 1)}{\myDots} × \correctionOr{(x + 1)}{\myDots}$
		\item $4x² - 12x + 9 = \correctionOr{(2x - 3)}{\myDots} × \correctionOr{(2x - 3)}{\myDots}$
		\item $9x² - 25 = \correctionOr{(3x + 5)}{\myDots} × \correctionOr{(3x - 2)}{\myDots}$
		\item $100x² + 80x + 16 = \correctionOr{(10x + 4)}{\myDots} × \correctionOr{(10x + 4)}{\myDots}$
		\item $2x(2x - 7) + 6x + 4 = \correctionOr{(2x - 2)}{\myDots} × \correctionOr{(2x - 2)}{\myDots}$
	\end{enumerate}
\end{exercice}
\vspace{1em}

\begin{exercice}
	Résoudre les équations suivantes : \medskip
	\begin{enumerate}
		\setlength{\itemsep}{1em}
		\item $x² = 25$ \correction{$x = 5$ ou $x = -5$}
		\item $x² + 3 = 19$ \correction{$x = 4$ ou $x = -4$}
		\item $(x + 2)(2x - 1) = 0$ \correction{$x = -2$ ou $x = 1/2$}
		\item $25x² - 49$ \correction{$x = 7/5$ ou $x = -7/5$}
		\item $36x² - 24x = -4$ \correction{$x = 1/3$}
	\end{enumerate}
\end{exercice}
}
