\begin{definition}[Variance, écart-type]
	Si on dispose d'une série de valeurs $x₁, ⋯, xₙ$, et qu'on dispose de la moyenne $M$, on définit
	\begin{itemize}
		\item La \textbf{variance} de cette série
		      $$ V = \dfrac{(x₁ - M)² + ⋯ + (xₙ - M)²}{n} $$
		\item L'\textbf{écart-type} de cette série
		      $$ σ = \sqrt{V} $$
	\end{itemize}
\end{definition}
}
