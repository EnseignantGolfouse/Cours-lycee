\newcommand{\Definition}{
	\begin{definition}[Univers, issues, évènements]
		\begin{itemize}
			\item Une \correctionDots{expérience aléatoire} est une expérience dont on ne peut pas prédire le résultat à l'avance.
			\item Lorsqu'on fait une expérience aléatoire, les résultats que l'on peut obtenir sont appelés les \correctionDots{issues}.
			\item L'ensemble complet des issues est appelé \correctionDots{l'univers}. On le note généralement $Ω$.
			\item Un ensemble d'issues est un \correctionDots{évènement}. On le note généralement avec une lettre majuscule, comme $A$, $B$, ...
			\item Si on a un évènement $A$, on appelle \correctionDots{$\overline{A}$ l'évènement contraire} de $A$ : c'est l'évènement qui contient les issues qui ne sont pas dans $A$.
		\end{itemize}
	\end{definition}
}
