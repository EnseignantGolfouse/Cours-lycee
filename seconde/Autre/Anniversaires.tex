\newcounter{xcounter}
\newcounter{ycounter}
\newcommand{\makeGrid}[2][]{%
	\setcounter{xcounter}{0}%
	\setcounter{ycounter}{0}%
	\begin{tikzpicture}[#1]
		\foreach \x in {0,...,4} {
				\draw (\x*2,0) -- (\x*2,8);
				\draw (0,\x*2) -- (8,\x*2);
			}
		\foreach \n in {#2} {
				\node at (2*\thexcounter + 1, 2*\theycounter + 1) {\n};

				\stepcounter{xcounter}
				\ifthenelse{\equal{\value{xcounter}}{4}}{
					\setcounter{xcounter}{0}
					\stepcounter{ycounter}
				}{}
			}
	\end{tikzpicture}
}
