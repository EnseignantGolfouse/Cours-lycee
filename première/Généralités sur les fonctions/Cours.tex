\begin{definition}[Fonction]
	Une \textbf{fonction} num√©rique est un proc√©d√© qui √† tout nombre associe un \textit{unique} autre nombre.

	La fonction est g√©n√©ralement not√©e $f$, le nombre de d√©part est not√© $x$ et le nombre obtenu est not√© $f(x)$. On le lit ¬´ $f$ de $x$ ¬ª, ou encore ¬´ $f$ appliqu√©e √† $x$ ¬ª.

	On la note
	$$ f : x ‚Ü¶ f(x) $$

	\begin{itemize}
		\item $f(x)$ est \textbf{l'image} de $x$ par la fonction $f$.

		      On repr√©sente une image par la lettre $y$, et on √©crit alors \uline{$f(x) = y$}.
		\item $x$ est \textbf{\uline{un} ant√©c√©dent} de $f(x)$ par la fonction $f$.
	\end{itemize}
\end{definition}

\begin{remarque}
	\begin{itemize}
		\item Pour un nombre donn√© $x$, il n'y a \uline{q'une seule image} $f(x)$.
		\item Pour un nombre donn√© $y$, il peut y avoir \uline{plusieurs ant√©c√©dents} $x$ tels que $y = f(x)$.
	\end{itemize}
\end{remarque}

\begin{definition}[Taux de variation]
	Soit $f$ une fonction, et $x‚ÇÅ$, $x‚ÇÇ$ deux nombres.

	Le \textbf{taux de variation} de la fonction $f$ entre $x‚ÇÅ$ et $x‚ÇÇ$ et donn√© par la formule

	$$ \dfrac{f(x‚ÇÇ) - f(x‚ÇÅ)}{x‚ÇÇ - x‚ÇÅ} $$

	Il correspond √† la pente de la droite $(d)$ suivante :

\begin{definition}[croissance, d√©croissance]
	Soit $f$ une fonction, et $I$ un intervalle de $‚Ñù$.

	\begin{itemize}
		\item On dit que $f$ est \textbf{croissante sur $I$} si pour tout nombres $x‚ÇÅ$ et $x‚ÇÇ$ dans $I$, le taux de variation $\dfrac{f(x‚ÇÇ) - f(x‚ÇÅ)}{x‚ÇÇ - x‚ÇÅ}$ est \textit{positif}.

	\end{itemize}
\end{definition}

\begin{definition}[Tableau de signes, de variations]
	Si $f$ est une fonction, on peut donner deux tableaux :
	\begin{itemize}

	\end{itemize}
\end{definition}

\section{Fonction affine}

\begin{definition}[Fonction affine]
	Une \textbf{fonction affine} est une fonction telle que $f(x) = ax + b$, avec $a$ et $b$ deux nombres r√©els.
\end{definition}

\begin{propriete}[Graphe d'une fonction affine]
	Le graphe d'un fonction affine est une \textbf{droite}, telle que :
	\begin{itemize}
		\item La pente de cette droite est $a$.
		\item La droite passe par le point $(0 ; b)$.
	\end{itemize}
\end{propriete}
