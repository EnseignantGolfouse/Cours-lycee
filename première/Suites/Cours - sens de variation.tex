\newcommand{\Propriete}{
	\begin{propriete}[Sens de variation]
		\begin{itemize}
			\item Si une suite est arithmétique de raison $r$, alors
			      \begin{itemize}
				      \item Si $\correctionDots{r > 0}$, la suite est \textbf{croissante}.
				      \item Si $\correctionDots{r = 0}$, la suite est \textbf{constante}.
				      \item Si $\correctionDots{r < 0}$, la suite est \textbf{décroissante}.
			      \end{itemize}
			\item Si une suite est géométrique de raison $q$, alors
			      \begin{itemize}
				      \item Si $\correctionDots{q > 1}$, la suite est \textbf{croissante}.
				      \item Si $\correctionDots{q = 1}$, la suite est \textbf{constante}.
				      \item Si $\correctionDots{q < 1}$, la suite est \textbf{décroissante}.
			      \end{itemize}
		\end{itemize}
	\end{propriete}
}

\Propriete
