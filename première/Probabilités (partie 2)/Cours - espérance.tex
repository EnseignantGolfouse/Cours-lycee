\begin{definition}[Espérance]
	Si la loi de probabilité de $X$ est donnée par
	\begin{center}
		\begin{tabular}{|c|c|c|c|c|}
			\hline
			$a_i$        & $a_1$ & $a_2$ & $⋯$ & $a_n$ \\ \hline
			$P(X = a_i)$ & $p_1$ & $p_2$ & $⋯$ & $p_n$ \\ \hline
		\end{tabular}
	\end{center}

	Alors son espérance est
	$$ E(X) = \correction{p_1a_1 + p_2a_2 + ⋯ + p_na_n} $$
\end{definition}
}
