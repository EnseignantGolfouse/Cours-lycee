		\item Remplir alors le tableau de signes suivant :

			      \tkzTabLine{, \correction{-}, \correctionOr{z}{}, , \correction{+}, , }
			      \tkzTabLine{, \correction{+}, \correctionOr{z}{}, \correction{-}, \correctionOr{z}{}, \correction{+}, }
		      \end{tikzpicture}
	\end{enumerate}
