	\begin{exercice}
		Déterminer :
		\begin{enumerate}
			\item $\{2 ; 3 ; 4\} ∪ \{1 ; 2; 3\} = \correctionDots{\{1 ; 2 ; 3 ; 4\}}$
			\item $\intervalle{]}{-1}{6}{]} ∩ \intervalle{[}{-2}{3}{]} = \correctionDots{\intervalle{]}{-1}{3}{]}}$
			\item $\intervalle{]}{3}{5}{]} ∪ \intervalle{[}{2}{3}{]} = \correctionDots{\intervalle{[}{2}{5}{]}}$
			\item $\intervalle{]}{-∞}{6}{[} ∩ \intervalle{[}{0}{7}{]} = \correctionDots{\intervalle{[}{0}{6}{[}}$
			\item $\intervalle{]}{-∞}{-2}{]} ∩ \intervalle{[}{0}{+∞}{]} = \correctionDots{∅\hspace{2em}}$
		\end{enumerate}
	\end{exercice}

	\begin{exercice}
		Pour chaque équation et inéquation ci-dessous, indiquer à quel ensemble appartient la solution :
		\begin{enumerate}
			\item Si $3x = 6$,\hspace{9.1em} $x ∈ \correctionDots{\{2\}\hspace{2em}}$
			\item Si $x ≤ 2$,\hspace{9.6em} $x ∈ \correctionDots{\intervalle{]}{-∞}{2}{]}}$
			\item Si $2x - 1 > 9$,\hspace{7.1em} $x ∈ \correctionDots{\intervalle{]}{5}{+∞}{[}}$
			\item Si $2x - 12 = 0$ ou $x - 2 = 8$, $x ∈ \correctionDots{\{6 ; 10\}\hspace{0.5em}}$
			\item Si $x ≤ 2$ et $x > -1$,\hspace{4.5em} $x ∈ \correctionDots{\intervalle{]}{-1}{2}{]}\hspace{0.3em}}$
		\end{enumerate}
	\end{exercice}

	\begin{exercice}
		\begin{enumerate}
			\item Résoudre l'équation $x × x = 0$.
			\item Soient $a$ et $b$ deux nombres. On cherche à résoudre $a × b = 0$.
			      \begin{itemize}
				      \item Si $a = 0$, alors $a × b$ $ \correctionDots{= 0}$
				      \item Si $b = 0$, alors $a × b$ $\correctionDots{= 0}$
				      \item Si $a ≠ 0$ et $b ≠ 0$, alors $a × b$ $\correctionDots{≠ 0}$
			      \end{itemize}

			      On a alors $a × b = 0$ si \correctionDots{$a = 0$ ou $b = 0$}.
			\item Résoudre l'équation $(x + 2) × (x - 1) = 0$ : \correction{$x ∈ \{-2 ; 1\}$}
		\end{enumerate}
	\end{exercice}
